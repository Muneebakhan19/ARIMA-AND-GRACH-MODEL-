import yfinance as yf
import pandas as pd
import numpy as np
from statsmodels.tsa.arima.model import ARIMA
from arch import arch_model
import warnings

# 1. CLEANUP: Ignore warnings for a clean output
warnings.filterwarnings("ignore")

# 2. DATA ACQUISITION: Using Adjusted Close for accuracy
ticker = 'AAPL' # Change to any ticker you need
data = yf.download(ticker, start='2018-01-01', end='2023-12-31', auto_adjust=True)

# 3. STATIONARITY: Calculate Log Returns and scale by 100
# This fixes the DataScaleWarning from your screenshots
data['Returns'] = 100 * np.log(data['Close'] / data['Close'].shift(1))
returns = data['Returns'].dropna()

# 4. OPTIMIZATION: Grid Search for the best ARIMA order (p, 0, q)
best_aic = float("inf")
best_order = (1, 0, 1) # Default fallback

print("Optimizing model... please wait.")
for p in range(3):
    for q in range(3):
        try:
            # We use d=0 because returns are already stationary
            model = ARIMA(returns, order=(p, 0, q)).fit()
            if model.aic < best_aic:
                best_aic = model.aic
                best_order = (p, 0, q)
        except:
            continue

# 5. FINAL MODEL FITTING
# Fitting ARIMA for the Mean (Price direction)
final_arima = ARIMA(returns, order=best_order).fit()

# Fitting GARCH(1,1) for Volatility (Risk) using Student's t-distribution
final_garch = arch_model(final_arima.resid, p=1, q=1, vol='GARCH', dist='t').fit(disp='off')

# 6. FORECASTING: Next 5 Business Days
arima_forecast = final_arima.get_forecast(steps=5)
mean_forecast = arima_forecast.predicted_mean

garch_forecast = final_garch.forecast(horizon=5)
# Convert variance to Volatility (Standard Deviation)
volatility_forecast = np.sqrt(garch_forecast.variance.values[-1])

# 7. CLEAN OUTPUT PRINTING
print(f"\n" + "="*45)
print(f"       5-DAY FORECAST FOR {ticker}")
print(f"   Optimized ARIMA Order: {best_order}")
print(f"   Best AIC Score: {best_aic:.2f}")
print("="*45)

# Generate future dates (skipping weekends)
last_date = data.index[-1]
forecast_dates = pd.date_range(start=last_date, periods=6, freq='B')[1:]

# Quick conversion for predicted prices for all 5 days
predicted_prices = [data['Close'].iloc[-1]] # Start with the last known closing price

for i in range(5):
    current_day_price = predicted_prices[-1]
    expected_return = mean_forecast.iloc[i] / 100
    next_day_price = current_day_price * (1 + expected_return)
    predicted_prices.append(next_day_price)

    print(f"Day {i+1} | {forecast_dates[i].date()}")
    print(f"  -> Expected Return:     {mean_forecast.iloc[i]:.4f}%")
    print(f"  -> Predicted Volatility: {volatility_forecast[i]:.4f}")
    print(f"  -> Predicted Price:     ${predicted_prices[i+1].iloc[0]:.2f}")
    print("-" * 45)
